class MetricScores():
  def __init__(self, y_teste):
    self.y_teste = y_teste
    self.reset_df()
    
  def reset_df(self):
    colunas = ["Classifier", "Precision", "Recall", "Accuracy", "F-Score"]
    dic = dict((k, []) for k in colunas)
    self.metric_df = pd.DataFrame(dic)
    
  def get_df(self):
    return self.metric_df
    
  def insert_prediction(self, classifier, predicao):
    precision = precision_score(self.y_teste, predicao)
    recall = recall_score(self.y_teste, predicao)
    accuracy = accuracy_score(self.y_teste, predicao)
    fscore = f1_score(self.y_teste, predicao)
    
    new_row = {
        "Accuracy": accuracy, "Classifier": classifier,
        "F-Score": fscore, "Precision": precision, "Recall": recall
    }
    
    self.metric_df.loc[1 + self.metric_df.shape[0]] = new_row

# @hidden_cell
def plot_learning_curve(estimator, title, X, y, ylim=None, cv=None,
                        n_jobs=None, train_sizes=np.linspace(.1, 1.0, 5)):
    # Funo importada dos exemplos do SKLearn
    plt.figure()
    plt.title(title)
    if ylim is not None:
        plt.ylim(*ylim)
    plt.xlabel(u"Training examples")
    plt.ylabel(u"Score")
    train_sizes, train_scores, test_scores = learning_curve(
        estimator, X, y, cv=cv, n_jobs=n_jobs, train_sizes=train_sizes)
    train_scores_mean = np.mean(train_scores, axis=1)
    train_scores_std = np.std(train_scores, axis=1)
    test_scores_mean = np.mean(test_scores, axis=1)
    test_scores_std = np.std(test_scores, axis=1)
    plt.grid()

    plt.fill_between(train_sizes, train_scores_mean - train_scores_std,
                     train_scores_mean + train_scores_std, alpha=0.1,
                     color="r")
    plt.fill_between(train_sizes, test_scores_mean - test_scores_std,
                     test_scores_mean + test_scores_std, alpha=0.1, color="g")
    plt.plot(train_sizes, train_scores_mean, 'o-', color="r",
             label=u"Training score")
    plt.plot(train_sizes, test_scores_mean, 'o-', color="g",
             label=u"Cross-validation score")

    plt.legend(loc="best")
    return plt

def plot_confusion_matrix(cm, classes, normalize=False, title='Confusion matrix',
                          cmap=plt.cm.Blues):
    # Funo importada dos exemplos do SKLearn
    if normalize:
        cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]

    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title)
    plt.colorbar()
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes, rotation=45)
    plt.yticks(tick_marks, classes)

    fmt = '.2f' if normalize else 'd'
    thresh = cm.max() / 2.
    for i, j in itertools.product(range(cm.shape[0]), range(cm.shape[1])):
        plt.text(j, i, format(cm[i, j], fmt),
                 horizontalalignment="center",
                 color="white" if cm[i, j] > thresh else "black")

    plt.ylabel(u'True label')
    plt.xlabel(u'Predicted label')
    plt.tight_layout()

def is_numerico(coluna):
  valor = coluna.ravel()[0]
  numericos = [np.float64, np.float32, np.int64]
  for n in numericos:
    if isinstance(valor, n):
      return True
  return False
  
  
def analisa_numericas(df):
  plt.figure(figsize=(16,24))
  plt.title('Anlise das variveis')

  col_numericas = []
  col_nao_normais = []
  alpha = 1e-3
  
  for coluna in list(df):
    if is_numerico(df[coluna]) and coluna != "ano" and coluna != "sequencial_candidato":
      col_numericas.append(coluna)
      if normaltest(df[coluna])[1] < alpha:
        col_nao_normais.append(coluna)
        
  for i in range(len(col_numericas)):
    coluna = col_numericas[i]
    plt.subplot(round(len(col_numericas)/2.0)+1, 2, i+1)
    plt.title(coluna)
    df[coluna].hist()
  plt.show()
  
  return col_nao_normais
  
colunas = analisa_numericas(treino)

def normaliza_dados(df, colunas, analisa=True):
  for col in colunas:
    temp = df[col][:].apply(np.log).values
    temp[temp == np.inf] = 0
    temp[temp == -np.inf] = 0
    temp[temp == np.nan] = 0
    df[col] = temp
  if (analisa):
    analisa_numericas(df)
    
  return df

treino = normaliza_dados(treino, colunas)
teste = normaliza_dados(teste, colunas, analisa = False)

  fatores = ['ano', 'uf', 'partido', 'cargo', 'sexo', 'estado_civil']

def codifica(df, col_nome, mapa):
  nova_coluna = []

  for i, row in df.iterrows():
    nova_coluna.append(mapa[row[col_nome]])

  df[col_nome] = nova_coluna
  return df

def codifica_escolaridade(df):
  niv_escolaridade = {'L E ESCREVE': 0, 'ENSINO FUNDAMENTAL INCOMPLETO': 1,
                      'ENSINO FUNDAMENTAL COMPLETO': 2,
                      'ENSINO MDIO INCOMPLETO': 3, 'ENSINO MDIO COMPLETO': 4,
                      'SUPERIOR INCOMPLETO': 5, 'SUPERIOR COMPLETO': 6 }
  return codifica(df, 'grau', niv_escolaridade)

def codifica_situacao(df):
  situacao = { 'eleito': 1, 'nao_eleito': 0 }
  return codifica(df, 'situacao', situacao)

treino = codifica_situacao(codifica_escolaridade(treino))

Como possvel ver no grfico acima, existe muito mais dados de deputados no eleitos do que de deputados eleitos. Para tratar esse problema existem duas possveis tcnicas: undersampling e oversampling. O undersampling consiste em reduzir a quantidade de amostras da categoria mais frequente ponto de ambas categorias ficarem com quantidade similar. O oversampling faz justamente o contrrio: seleciona amostras aleatrias dos dados que tem menor frequncia e adiciona duplicatas nos dados, tentando igualar a frequncia das classes.

Como retirar dados parece ser pior do que adicionar novos dados, a estratgia que eu escolhi o oversampling:

Treinando Árvores de Decisão¶

Para treinar as rvores, configuramos alguns parmetros e selecionamos os que se desempenham melhor:

  • Nmero mnimo de elementos por folha, entre 1, 5, 10, 15, 20, 30, 50, 75, 100.
  • Profundidade mxima, entre 1, 5, 10, 15, 25, 30, 50 ou sem limite.
  • Nmero mximo de folhas, entre 50, 40, 30, 20, 10, 5 ou sem limite.
  • Nmero mximo de features, escolhido por sqrt, log2 ou sem limite mximo.
  • Nmero mnimo de elementos ao dividir um n, entre 2, 5, 10, 15, 20, 25, 30.

Treinando o AdaBoost¶

Para treinar o AdaBoost, escolheremos um modelo de classificao fraco: uma rvore de deciso com ltura mxima sendo 1. Isso fora a simplicidade do modelo e ajuda o algoritmo de boosting por ser um classificador fraco. Alm disso, se define o hiper-parmetro T, que ser o nmero mximo de estimadores, atravs do tuning.

  pred = ada.predict(x_teste.drop(seq, axis=1))
  fscore = f1_score(y_teste, pred)

    best_ada = [(t, alg), fscore, pred]

Stacking¶

Para implementar o stacking, inicialmente instanciaremos classificadores treinados com todo o conjunto de dados. Em seguida, prediremos o conjunto de testes e faremos um dataframe com as predies de cada um dos classificadores, e o usaremos para obter a predio do conjunto de testes
